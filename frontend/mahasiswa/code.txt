        const user = initCommon('MAHASISWA');
        let allLowongan = [];

        // =============================================
        // TOAST — pengganti SweetAlert
        // =============================================
        const ICONS = { success: '✅', error: '❌', warning: '⚠️', info: 'ℹ️' };

        function showToastCustom(title, message, type = 'info', duration = 4000) {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.style.position = 'relative';
            toast.style.overflow = 'hidden';
            toast.innerHTML = `
                <div class="toast-icon">${ICONS[type] || 'ℹ️'}</div>
                <div class="toast-body">
                    <div class="toast-title">${title}</div>
                    ${message ? `<div class="toast-msg">${message}</div>` : ''}
                </div>
                <button class="toast-close" onclick="this.parentElement.remove()">×</button>
                <div class="toast-progress" style="animation-duration: ${duration}ms;"></div>
            `;
            container.appendChild(toast);
            setTimeout(() => {
                toast.style.animation = 'slideOut 0.3s ease forwards';
                setTimeout(() => toast.remove(), 300);
            }, duration);
        }

        // =============================================
        // CONFIRM DIALOG — pengganti SweetAlert confirm
        // =============================================
        function showConfirm({ icon = '❓', title, text, okText = 'Ya', cancelText = 'Batal', okDanger = false }) {
            return new Promise((resolve) => {
                const overlay = document.getElementById('confirm-overlay');
                document.getElementById('confirm-icon').textContent  = icon;
                document.getElementById('confirm-title').textContent = title;
                document.getElementById('confirm-text').textContent  = text;

                const btnOk     = document.getElementById('confirm-btn-ok');
                const btnCancel = document.getElementById('confirm-btn-cancel');
                btnOk.textContent     = okText;
                btnCancel.textContent = cancelText;
                btnOk.className = okDanger ? 'danger' : '';

                overlay.style.display = 'flex';

                // Hapus listener lama
                const newOk     = btnOk.cloneNode(true);
                const newCancel = btnCancel.cloneNode(true);
                btnOk.parentNode.replaceChild(newOk, btnOk);
                btnCancel.parentNode.replaceChild(newCancel, btnCancel);

                document.getElementById('confirm-btn-ok').addEventListener('click', () => {
                    overlay.style.display = 'none';
                    resolve(true);
                });
                document.getElementById('confirm-btn-cancel').addEventListener('click', () => {
                    overlay.style.display = 'none';
                    resolve(false);
                });
            });
        }

        // =============================================
        // HELPER: Error CV inline
        // =============================================
        function showCvError() {
            document.getElementById('cv-dropzone').classList.add('dropzone-error');
            document.getElementById('cv-link').classList.add('input-error');
            document.getElementById('cv-error-msg').classList.add('show');
            document.getElementById('cv-section').scrollIntoView({ behavior: 'smooth', block: 'center' });
        }

        function clearCvError() {
            document.getElementById('cv-dropzone').classList.remove('dropzone-error');
            document.getElementById('cv-link').classList.remove('input-error');
            document.getElementById('cv-error-msg').classList.remove('show');
        }

        // =============================================
        // Handle file selection
        // =============================================
        function handleFileSelect(input) {
            const file = input.files[0];
            if (!file) return;

            if (file.type !== 'application/pdf') {
                showToastCustom('Format Tidak Valid', 'Hanya file PDF yang diperbolehkan!', 'error', 5000);
                input.value = '';
                return;
            }
            if (file.size > 5 * 1024 * 1024) {
                showToastCustom('File Terlalu Besar', 'Ukuran file maksimal 5MB!', 'error', 5000);
                input.value = '';
                return;
            }

            selectedFile = file;
            document.getElementById('cv-upload-area').classList.add('hidden');
            document.getElementById('cv-file-info').classList.remove('hidden');
            document.getElementById('cv-file-name').textContent = file.name;
            document.getElementById('cv-link').value = '';
            clearCvError();
        }

        function clearFile() {
            selectedFile = null;
            document.getElementById('cv-file').value = '';
            document.getElementById('cv-upload-area').classList.remove('hidden');
            document.getElementById('cv-file-info').classList.add('hidden');
        }

        // =============================================
        // Fetch & display lowongan
        // =============================================
        async function fetchLowongan() {
            showLoading('lowongan-list');
            try {
                const response = await fetch(`${API_URL}/lowongan`, {
                    headers: { 'Authorization': `Bearer ${getToken()}` }
                });
                const data = await response.json();
                allLowongan = data;
                displayLowongan(data);
            } catch (error) {
                showError('lowongan-list', 'Gagal memuat lowongan');
            }
        }

        function displayLowongan(lowongan) {
            const container = document.getElementById('lowongan-list');
            if (lowongan.length === 0) {
                container.innerHTML = '<p class="col-span-full text-center text-gray-500">Tidak ada lowongan</p>';
                return;
            }
            container.innerHTML = lowongan.map(item => `
                <div class="bg-white rounded-lg shadow p-6 hover:shadow-lg transition cursor-pointer"
                    onclick="showDetail('${item.id}')">
                    <h3 class="font-bold text-lg mb-2">${item.judul}</h3>
                    <p class="text-gray-600 text-sm mb-2">${item.perusahaan.namaPerusahaan}</p>
                    <p class="text-gray-700 text-sm mb-3">${item.deskripsi.substring(0, 100)}...</p>
                    <div class="flex justify-between items-center text-sm">
                        <span class="text-gray-500">📍 ${item.lokasi}</span>
                        <span class="px-3 py-1 rounded ${getStatusClass(item.status)}">${item.status}</span>
                    </div>
                </div>
            `).join('');
        }

        // =============================================
        // Show detail modal
        // =============================================
        async function showDetail(id) {
            try {
                const response = await fetch(`${API_URL}/lowongan/${id}`, {
                    headers: { 'Authorization': `Bearer ${getToken()}` }
                });
                const lowongan = await response.json();

                document.getElementById('lowongan-id').value = id;
                document.getElementById('modal-title').textContent = lowongan.judul;
                document.getElementById('modal-content').innerHTML = `
                    <div class="space-y-4">
                        <div class="bg-blue-50 p-4 rounded-lg">
                            <h4 class="font-semibold text-blue-900">${lowongan.perusahaan.namaPerusahaan}</h4>
                            <p class="text-sm text-blue-700 mt-1">${lowongan.lokasi}</p>
                        </div>
                        <div>
                            <h4 class="font-semibold mb-2">Deskripsi</h4>
                            <p class="text-gray-700">${lowongan.deskripsi}</p>
                        </div>
                        <div>
                            <h4 class="font-semibold mb-2">Persyaratan</h4>
                            <p class="text-gray-700 whitespace-pre-wrap">${lowongan.persyaratan}</p>
                        </div>
                        <div class="grid grid-cols-2 gap-4 bg-gray-50 p-4 rounded-lg">
                            <div>
                                <span class="text-sm text-gray-600">Durasi</span>
                                <p class="font-medium">${lowongan.durasi}</p>
                            </div>
                            <div>
                                <span class="text-sm text-gray-600">Kuota</span>
                                <p class="font-medium">${lowongan.kuota} orang</p>
                            </div>
                            <div>
                                <span class="text-sm text-gray-600">Periode Lamaran</span>
                                <p class="font-medium text-sm">
                                    ${new Date(lowongan.tanggalMulai).toLocaleDateString('id-ID')} -
                                    ${new Date(lowongan.tanggalSelesai).toLocaleDateString('id-ID')}
                                </p>
                            </div>
                            <div>
                                <span class="text-sm text-gray-600">Status</span>
                                <p class="font-medium">
                                    <span class="px-2 py-1 rounded text-xs ${lowongan.status === 'BUKA' ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'}">
                                        ${lowongan.status}
                                    </span>
                                </p>
                            </div>
                        </div>
                    </div>
                `;

                document.getElementById('apply-form').reset();
                clearFile();
                clearCvError();
                isSubmitting = false;
                document.getElementById('apply-btn').disabled = false;
                document.getElementById('apply-btn').textContent = 'Kirim Lamaran';

                document.getElementById('detail-modal').classList.remove('hidden');
            } catch (error) {
                showToastCustom('Gagal Memuat', 'Gagal memuat detail lowongan. Silakan coba lagi.', 'error', 5000);
            }
        }

        document.getElementById('close-modal').addEventListener('click', () => {
            document.getElementById('detail-modal').classList.add('hidden');
        });

        // =============================================
        // SUBMIT LAMARAN
        // =============================================
        document.getElementById('apply-btn').addEventListener('click', async () => {
            if (isSubmitting) return;

            const cvFile = selectedFile;
            const cvLink = document.getElementById('cv-link').value.trim();

            // STEP 1 — Validasi CV
            if (!cvFile && !cvLink) {
                showCvError();
                showToastCustom('Data Belum Lengkap', 'CV file atau link CV harus diisi sebelum mengirim lamaran.', 'warning', 6000);
                return;
            }

            clearCvError();

            // STEP 2 — Konfirmasi
            const ok = await showConfirm({
                icon: '📋',
                title: 'Kirim Lamaran?',
                text: 'Pastikan semua data lamaran Anda sudah benar sebelum dikirim.',
                okText: 'Ya, Kirim!',
                cancelText: 'Cek Lagi'
            });

            if (!ok) return;

            // STEP 3 — Loading
            isSubmitting = true;
            const btn = document.getElementById('apply-btn');
            btn.disabled = true;
            btn.textContent = 'Mengirim...';

            // STEP 4 — Kirim ke server
            const formData = new FormData();
            formData.append('lowonganId', document.getElementById('lowongan-id').value);
            if (cvFile) formData.append('cv', cvFile);
            if (cvLink) formData.append('cvLink', cvLink);

            const portfolioLink = document.getElementById('portfolio-link').value;
            const linkedinLink  = document.getElementById('linkedin-link').value;
            const resumeLink    = document.getElementById('resume-link').value;
            if (portfolioLink) formData.append('portfolioLink', portfolioLink);
            if (linkedinLink)  formData.append('linkedinLink', linkedinLink);
            if (resumeLink)    formData.append('resumeLink', resumeLink);

            try {
                const response = await fetch(`${API_URL}/mahasiswa/lamaran`, {
                    method: 'POST',
                    headers: { 'Authorization': `Bearer ${getToken()}` },
                    body: formData
                });

                let result;
                const contentType = response.headers.get("content-type");

                if (contentType && contentType.includes("application/json")) {
                    result = await response.json();
                } else {
                    result = { message: await response.text() };
                }

                if (!response.ok) {
                    // const errData = await response.json();
                    // const text = await response.text();
                    // console.error("SERVER RESPONSE:", text);
                    throw new Error(result.message || 'Gagal mengirim lamaran');
                }

                // STEP 5 — Sukses
                document.getElementById('detail-modal').classList.add('hidden');
                document.getElementById('apply-form').reset();
                clearFile();
                clearCvError();

                // Toast sukses tampil 8 detik
                showToastCustom(
                    'Lamaran Berhasil Dikirim! 🎉',
                    'Lamaran Anda telah berhasil dikirim. Pantau statusnya di menu "Lamaran Saya".',
                    'success',
                    8000
                );

            } catch (error) {
                // Toast error tampil 8 detik
                showToastCustom(
                    'Gagal Mengirim Lamaran',
                    error.message || 'Terjadi kesalahan. Silakan coba lagi.',
                    'error',
                    8000
                );
            } finally {
                isSubmitting = false;
                btn.disabled = false;
                btn.textContent = 'Kirim Lamaran';
            }
        });

        // =============================================
        // Search & Filter
        // =============================================
        document.getElementById('search').addEventListener('input', filterLowongan);
        document.getElementById('filter-status').addEventListener('change', filterLowongan);

        function filterLowongan() {
            const search = document.getElementById('search').value.toLowerCase();
            const status = document.getElementById('filter-status').value;
            const filtered = allLowongan.filter(item => {
                const matchSearch = item.judul.toLowerCase().includes(search) ||
                                    item.perusahaan.namaPerusahaan.toLowerCase().includes(search);
                const matchStatus = !status || item.status === status;
                return matchSearch && matchStatus;
            });
            displayLowongan(filtered);
        }
